bool leptonReadRegister(uint16_t regAddr, uint16_t* result) {
  Wire.beginTransmission(0x2A);
  Wire.write(highByte(regAddr));
  Wire.write(lowByte(regAddr));
  if (Wire.endTransmission(false) != 0) return false;  // Restart condition

  Wire.requestFrom(0x2A, 2);  // Read 2 bytes
  if (Wire.available() < 2) return false;
  
  *result = (Wire.read() << 8) | Wire.read();  // Combine MSB+LSB
  return true;

UnitLepton lepton;
AsyncWebServer server(80);

uint8_t* gray_buffer;
size_t jpeg_size = 0;
uint8_t* jpeg_buffer = nullptr;

// WiFi credentials
const char* ssid = "your_wifi_ssid";
const char* password = "your_wifi_password";

void setup() {
    M5.begin();
    lepton.begin();

    // Initialize WiFi
    WiFi.begin(ssid, password);
    while (WiFi.status() != WL_CONNECTED) {
        delay(500);
    }
    M5.Lcd.printf("IP Address: %s\n", WiFi.localIP().toString().c_str());

    // Allocate buffers in PSRAM
    gray_buffer = (uint8_t*)heap_caps_malloc(80 * 60, MALLOC_CAP_SPIRAM);

    // Configure JPEG encoder
    esp_jpeg_config_t jpeg_cfg = {
        .input_format = JPEG_INPUT_FORMAT_GRAY,
        .input_width = 80,
        .input_height = 60,
        .quality = 60
    };
    esp_jpeg_init(&jpeg_cfg);

    // Setup MJPEG stream endpoint
    server.on("/stream", HTTP_GET, [](AsyncWebServerRequest *request){
        request->send(200, "multipart/x-mixed-replace; boundary=frame");
    });

    // Start server
    server.begin();
}

void loop() {
    lepton.update();

    // Convert raw Lepton data to 8-bit grayscale
    for (int i = 0; i < 80 * 60; i++) {
        int16_t raw_value = *(lepton.getRawValues() + i);
        // Normalize to 0-255 with auto-ranging
        gray_buffer[i] = map(raw_value, lepton.getMinValue(), lepton.getMaxValue(), 0, 255);
    }

    // Encode to JPEG
    jpeg_size = esp_jpeg_encode(gray_buffer, 80 * 60, &jpeg_buffer);

    // Send JPEG over MJPEG stream
    if (jpeg_size > 0 && jpeg_buffer != nullptr) {
        server.client().print("HTTP/1.1 200 OK\r\n");
        server.client().print("Content-Type: image/jpeg\r\n");
        server.client().printf("Content-Length: %d\r\n", jpeg_size);
        server.client().print("Connection: close\r\n\r\n");
        server.client().write(jpeg_buffer, jpeg_size);
        server.client().flush();
    }

    // Cleanup
    if (jpeg_buffer != nullptr) {
        free(jpeg_buffer);
        jpeg_buffer = nullptr;
    }

    delay(100); // 10fps stream rate
}

#include <M5StickT2.h>
#include <WiFi.h>
#include <Lepton.h>
#include <WebServer.h>

const char* ssid = "YOUR_WIFI_SSID";
const char* password = "YOUR_WIFI_PASSWORD";

WebServer server(80);
Lepton myLepton;
uint8_t thermalFrame[80 * 60 * 2]; // Buffer for 14-bit thermal data

// Color mapping for thermal visualization
uint16_t colorPalette[256];

void generateColorPalette() {
  // Generate ironbow color palette for thermal imaging
  for (int i = 0; i < 256; i++) {
    uint8_t r = (i < 128) ? map(i, 0, 128, 0, 255) : 255;
    uint8_t g = (i < 128) ? 0 : map(i, 128, 255, 0, 255);
    uint8_t b = map(i, 0, 255, 255, 0);
    colorPalette[i] = ((r & 0xF8) << 8) | ((g & 0xFC) << 3) | (b >> 3);
  }
}

void handleMJPEG() {
  server.setContentLength(CONTENT_LENGTH_UNKNOWN);
  server.send(200, "multipart/x-mixed-replace; boundary=frame");

  while (true) {
    // Capture thermal frame
    int res = myLepton.getFrame(thermalFrame);
    if (res != 0) continue;

    // Convert 14-bit thermal data to 8-bit and apply color mapping
    uint16_t rgbFrame[80 * 60];
    for (int i = 0; i < 80 * 60; i++) {
      uint16_t temp = (thermalFrame[i*2] << 8) | thermalFrame[i*2+1];
      uint8_t normalized = map(temp, 27315, 32315, 0, 255); // 0°C to 50°C range
      rgbFrame[i] = colorPalette[normalized];
    }

    // Encode to JPEG
    uint8_t jpegBuffer[80 * 60 * 2];
    size_t jpegSize = jpeg_encode(rgbFrame, 80, 60, 80*2, jpegBuffer, sizeof(jpegBuffer), 80);

    // Send MJPEG frame
    server.sendContent("--frame\r\n");
    server.sendContent("Content-Type: image/jpeg\r\n");
    server.sendContent("Content-Length: " + String(jpegSize) + "\r\n\r\n");
    server.sendContent((const char*)jpegBuffer, jpegSize);
    server.sendContent("\r\n");

    // Handle client disconnect
    if (!server.client().connected()) break;
  }
}

void setup() {
  M5.begin();
  Wire.begin(0, 26);
  Serial.begin(115200);

  myLepton.setTLinearMode(1);

  generateColorPalette();

  Serial.println("");

  Serial.println("IP address: ");
  Serial.println(WiFi.localIP());

  // Start web server
